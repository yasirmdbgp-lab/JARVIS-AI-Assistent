<script>
    // Language settings
    let currentLang = 'en'; // 'en' or 'hi'
    
    // Messages in both languages
    const messages = {
        en: {
            listening: "🎙️ Listening...",
            processing: "🧠 Processing...",
            ready: "Ready",
            thinking: "🤔 Thinking...",
            noInput: "I didn't hear anything. Please try again.",
            sorry: "Sorry, I couldn't understand. Please try again.",
            greeting: "Hello! I'm JARVIS. How can I help you?",
            greeting2: "Good to see you again! What can I do for you?",
            help: "I can answer questions, tell you the time, search the web, and more!"
        },
        hi: {
            listening: "🎙️ सुन रहा हूँ...",
            processing: "🧠 प्रोसेस कर रहा हूँ...",
            ready: "तैयार",
            thinking: "🤔 सोच रहा हूँ...",
            noInput: "मुझे कुछ सुनाई नहीं दिया। कृपया फिर से बोलें।",
            sorry: "माफ़ करें, मैं समझ नहीं पाया। कृपया फिर से कोशिश करें।",
            greeting: "नमस्ते! मैं JARVIS हूँ। आपकी कैसे मदद कर सकता हूँ?",
            greeting2: "आपको फिर देखकर अच्छा लगा! मैं क्या कर सकता हूँ आपके लिए?",
            help: "मैं सवालों के जवाब दे सकता हूँ, समय बता सकता हूँ, वेब सर्च कर सकता हूँ, और भी बहुत कुछ!"
        }
    };

    // DOM Elements
    const conversationDiv = document.getElementById('conversation');
    const jarvisCircle = document.getElementById('jarvisCircle');
    const jarvisStatus = document.getElementById('jarvisStatus');
    const voiceBtn = document.getElementById('voiceBtn');
    const textInput = document.getElementById('textInput');
    const sendBtn = document.getElementById('sendBtn');
    const langBtns = document.querySelectorAll('.lang-btn');

    // Speech Recognition
    let recognition = null;
    let isListening = false;

    // Speech Synthesis
    let synth = window.speechSynthesis;

    // Check browser support
    if ('webkitSpeechRecognition' in window || 'SpeechRecognition' in window) {
        const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
        recognition = new SpeechRecognition();
        recognition.continuous = false;
        recognition.interimResults = false;
        recognition.maxAlternatives = 1;
        
        // Set language based on current selection
        recognition.lang = currentLang === 'en' ? 'en-US' : 'hi-IN';
    } else {
        addMessage("⚠️ Your browser doesn't support voice recognition. Please use Chrome or Edge on mobile.", 'jarvis-message');
        voiceBtn.style.opacity = '0.5';
        voiceBtn.style.pointerEvents = 'none';
    }

    // Add message to conversation
    function addMessage(text, className) {
        const messageDiv = document.createElement('div');
        messageDiv.className = `message ${className}`;
        
        // Add emoji prefix
        if (className === 'jarvis-message') {
            messageDiv.innerHTML = `🤖 ${text}`;
        } else {
            messageDiv.innerHTML = `👤 ${text}`;
        }
        
        conversationDiv.appendChild(messageDiv);
        conversationDiv.scrollTop = conversationDiv.scrollHeight;
    }

    // Speak text
    function speak(text) {
        if (!synth) return;
        
        // Cancel any ongoing speech
        synth.cancel();
        
        const utterance = new SpeechSynthesisUtterance(text);
        
        // Set language
        if (currentLang === 'hi') {
            utterance.lang = 'hi-IN';
            // Use Hindi voice if available
            const voices = synth.getVoices();
            const hindiVoice = voices.find(voice => voice.lang === 'hi-IN');
            if (hindiVoice) utterance.voice = hindiVoice;
        } else {
            utterance.lang = 'en-US';
        }
        
        utterance.rate = 0.9;
        utterance.pitch = 1.0;
        
        utterance.onend = () => {
            if (!isListening) {
                jarvisStatus.innerText = messages[currentLang].ready;
                jarvisCircle.classList.remove('listening');
            }
        };
        
        synth.speak(utterance);
    }

    // Process user input and generate response
    async function processInput(userText) {
        if (!userText || userText.trim() === '') {
            addMessage(messages[currentLang].noInput, 'jarvis-message');
            speak(messages[currentLang].noInput);
            return;
        }
        
        // Show thinking status
        jarvisStatus.innerText = messages[currentLang].thinking;
        addMessage(userText, 'user-message');
        
        // Convert to lowercase for easier matching
        const lowerText = userText.toLowerCase();
        
        // --- Smart Response System (No API Key Required) ---
        let response = '';
        
        // Time related queries
        if (lowerText.includes('time') || lowerText.includes('टाइम') || lowerText.includes('समय') || lowerText.includes('बजे')) {
            const now = new Date();
            const timeStr = now.toLocaleTimeString(currentLang === 'en' ? 'en-US' : 'hi-IN', { hour: '2-digit', minute: '2-digit' });
            response = currentLang === 'en' ? `The time is ${timeStr}` : `समय ${timeStr} बज रहा है`;
        }
        
        // Date queries
        else if (lowerText.includes('date') || lowerText.includes('डेट') || lowerText.includes('तारीख') || lowerText.includes('आज')) {
            const now = new Date();
            const dateStr = now.toLocaleDateString(currentLang === 'en' ? 'en-US' : 'hi-IN', { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' });
            response = currentLang === 'en' ? `Today is ${dateStr}` : `आज ${dateStr} है`;
        }
        
        // Greetings
        else if (lowerText.includes('hello') || lowerText.includes('hi') || lowerText.includes('नमस्ते') || lowerText.includes('हेलो')) {
            response = messages[currentLang].greeting;
        }
        
        // How are you
        else if (lowerText.includes('how are you') || lowerText.includes('कैसे हो') || lowerText.includes('कैसा है')) {
            response = currentLang === 'en' ? "I'm doing great, sir! Always ready to serve you. How can I help?" : "मैं बहुत बढ़िया हूँ, सर! हमेशा आपकी सेवा के लिए तैयार। आप कैसे हैं?";
        }
        
        // What can you do / Help
        else if (lowerText.includes('what can you do') || lowerText.includes('help') || lowerText.includes('मदद') || lowerText.includes('क्या कर सकते हो')) {
            response = messages[currentLang].help;
        }
        
        // Weather (demo)
        else if (lowerText.includes('weather') || lowerText.includes('मौसम')) {
            response = currentLang === 'en' ? "I can't access live weather without internet. Please check your weather app!" : "मैं बिना इंटरनेट के लाइव मौसम नहीं देख सकता। कृपया अपना वेदर ऐप चेक करें!";
        }
        
        // Name
        else if (lowerText.includes('your name') || lowerText.includes('तुम्हारा नाम') || lowerText.includes('कौन हो')) {
            response = currentLang === 'en' ? "I am JARVIS, your personal AI voice assistant. Ready to help 24/7!" : "मैं JARVIS हूँ, आपका निजी AI वॉइस असिस्टेंट। 24/7 मदद के लिए तैयार!";
        }
        
        // Thank you
        else if (lowerText.includes('thank') || lowerText.includes('धन्यवाद') || lowerText.includes('शुक्रिया')) {
            response = currentLang === 'en' ? "You're most welcome, sir! Always happy to help." : "आपका स्वागत है, सर! मदद करके हमेशा खुशी होती है।";
        }
        
        // Search web
        else if (lowerText.includes('search') || lowerText.includes('सर्च') || lowerText.includes('ढूंढो')) {
            let searchTerm = userText.replace(/search|सर्च|ढूंढो/gi, '').trim();
            if (searchTerm) {
                response = currentLang === 'en' ? `Searching for "${searchTerm}" on Google.` : `Google पर "${searchTerm}" खोज रहा हूँ।`;
                setTimeout(() => {
                    window.open(`https://www.google.com/search?q=${encodeURIComponent(searchTerm)}`, '_blank');
                }, 1000);
            } else {
                response = currentLang === 'en' ? "What would you like me to search for?" : "आप क्या खोजना चाहेंगे?";
            }
        }
        
        // Open YouTube
        else if (lowerText.includes('youtube') || lowerText.includes('यूट्यूब') || lowerText.includes('video')) {
            response = currentLang === 'en' ? "Opening YouTube for you, sir!" : "आपके लिए YouTube खोल रहा हूँ, सर!";
            setTimeout(() => {
                window.open('https://m.youtube.com', '_blank');
            }, 1000);
        }
        
        // Joke
        else if (lowerText.includes('joke') || lowerText.includes('चुटकुला') || lowerText.includes('मजाक')) {
            const jokes = currentLang === 'en' 
                ? ["Why don't scientists trust atoms? Because they make up everything!",
                   "What do you call a fake noodle? An impasta!",
                   "Why did the scarecrow win an award? He was outstanding in his field!"]
                : ["एक आदमी ने कहा - मेरी बीवी मुझसे बहुत प्यार करती है! दूसरे ने पूछा - कैसे पता? पहला बोला - जब भी मैं कहीं जाता हूँ, वो मुझे हैंडग्रेनैड देकर भेजती है ताकि कोई मुझे छू भी न सके!",
                   "टीचर: तुम्हारे पिताजी क्या करते हैं? छात्र: जो भी माँ कहती हैं!",
                   "डॉक्टर: आपको दूध के साथ दवा लेनी है। मरीज: ठीक है, दूध कब लाना है?"];
            response = jokes[Math.floor(Math.random() * jokes.length)];
        }
        
        // AI / GPT-like response for other queries
        else {
            const defaultResponses = {
                en: [
                    "Interesting! Tell me more about that.",
                    "I understand. How can I assist you further?",
                    "That's great! Is there anything specific you'd like to know?",
                    "I'm here to help. What else can I do for you?"
                ],
                hi: [
                    "दिलचस्प! इसके बारे में और बताइए।",
                    "मैं समझ गया। मैं और कैसे मदद कर सकता हूँ?",
                    "यह बहुत अच्छा है! क्या आप कुछ विशेष जानना चाहेंगे?",
                    "मैं यहाँ मदद करने के लिए हूँ। आपके लिए और क्या कर सकता हूँ?"
                ]
            };
            response = defaultResponses[currentLang][Math.floor(Math.random() * defaultResponses[currentLang].length)];
        }
        
        // Add response and speak
        setTimeout(() => {
            addMessage(response, 'jarvis-message');
            speak(response);
            jarvisStatus.innerText = messages[currentLang].ready;
            jarvisCircle.classList.remove('listening');
        }, 500);
    }

    // Start voice recognition
    function startListening() {
        if (!recognition) {
            addMessage("Voice recognition not supported. Please type your message.", 'jarvis-message');
            return;
        }
        
        if (isListening) return;
        
        try {
            // Set language
            recognition.lang = currentLang === 'en' ? 'en-US' : 'hi-IN';
            
            isListening = true;
            jarvisCircle.classList.add('listening');
            jarvisStatus.innerText = messages[currentLang].listening;
            
            recognition.start();
            
            recognition.onresult = (event) => {
                const transcript = event.results[0][0].transcript;
                jarvisStatus.innerText = messages[currentLang].processing;
                processInput(transcript);
                isListening = false;
                jarvisCircle.classList.remove('listening');
            };
            
            recognition.onerror = (event) => {
                console.error('Recognition error:', event.error);
                jarvisStatus.innerText = messages[currentLang].ready;
                jarvisCircle.classList.remove('listening');
                isListening = false;
                
                if (event.error === 'not-allowed') {
                    addMessage("Please allow microphone access to use voice features.", 'jarvis-message');
                } else if (event.error !== 'no-speech') {
                    addMessage(messages[currentLang].sorry, 'jarvis-message');
                }
            };
            
            recognition.onend = () => {
                isListening = false;
                if (jarvisStatus.innerText === messages[currentLang].listening) {
                    jarvisStatus.innerText = messages[currentLang].ready;
                    jarvisCircle.classList.remove('listening');
                }
            };
            
        } catch (e) {
            console.error(e);
            jarvisStatus.innerText = messages[currentLang].ready;
            jarvisCircle.classList.remove('listening');
            isListening = false;
        }
    }

    // Change language
    function setLanguage(lang) {
        currentLang = lang;
        
        // Update button styles
        langBtns.forEach(btn => {
            if (btn.dataset.lang === lang) {
                btn.classList.add('active');
            } else {
                btn.classList.remove('active');
            }
        });
        
        // Update recognition language if available
        if (recognition) {
            recognition.lang = currentLang === 'en' ? 'en-US' : 'hi-IN';
        }
        
        jarvisStatus.innerText = messages[currentLang].ready;
        
        // Add language change message
        const langMsg = currentLang === 'en' ? "Switched to English. You can speak in English now!" : "हिंदी में स्विच किया। अब आप हिंदी में बोल सकते हैं!";
        addMessage(langMsg, 'jarvis-message');
        speak(langMsg);
    }

    // Event Listeners
    voiceBtn.addEventListener('click', startListening);
    
    sendBtn.addEventListener('click', () => {
        const userText = textInput.value.trim();
        if (userText) {
            processInput(userText);
            textInput.value = '';
        }
    });
    
    textInput.addEventListener('keypress', (e) => {
        if (e.key === 'Enter') {
            sendBtn.click();
        }
    });
    
    langBtns.forEach(btn => {
        btn.addEventListener('click', () => {
            setLanguage(btn.dataset.lang);
        });
    });
    
    // Optional: Click on JARVIS circle to listen
    jarvisCircle.addEventListener('click', startListening);
    
    // Welcome message with voice
    setTimeout(() => {
        speak(messages[currentLang].greeting);
    }, 500);
</script>
